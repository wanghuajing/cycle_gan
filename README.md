### 训练生成256*256分辨率的图片参数
	--dataroot
	/home/zhao/mydata/datasets/mammo300/
	--name
	pix2pix_256
	--model
	pix2pix_256
	--input_nc
	1
	--output_nc
	1
	--netD
	n_layers
	--n_layers_D
	6
	--netG
	resnet_9blocks
	--preprocess
	resize
	--no_flip
	--crop_size
	256
	--batch_size
	1
